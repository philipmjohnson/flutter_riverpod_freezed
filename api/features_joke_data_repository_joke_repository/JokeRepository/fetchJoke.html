Future<JokeModel> fetchJoke() async {
  try {
    final res = await _jokeApi.fetchJokesApiRequest();
    final jokeModel = JokeModel.fromJson(res);
    return jokeModel;
  } on DioError catch (e) {
    final errorMessage = DioExceptions.fromDioError(e);
    log(errorMessage.toString());
    rethrow;
  }
}
